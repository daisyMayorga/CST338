M2: Casino Java Program
08 May 2017

PURPOSE:
The purpose of this program is to design a casino slot machine game.

OVERVIEW:
The program will loop, asking the user for a bet amount from 0 to 100 (assume dollars, you can use ints or longs).
If the user types a 0 that means she wants to quit.  Otherwise, accept the amount as their bet
and simulate a slot machine pull.  Your program will print out a line that looks like a slot machine result
containing three strings.  Some examples are:

BAR  7  BAR, 7  7  cherries, cherries  BAR  space, space  BAR  BAR, or cherries  cherries  BAR.

Each of the three positions in the string could be one of the following:  "BAR", "7", "cherries" or "space". 
Each of the three output positions must be generated by your program randomly with probabilities:

BAR          1/2  (50%)
cherries     1/4  (25%)
space        1/8  (12.5%)
7            1/8  (12.5%)

Therefore, BAR should be the most frequent symbol seen and space or 7 the least frequent.
The following combinations should pay the bet as shown (note ORDER MATTERS):

cherries  [not cherries]  [any] pays 5 × bet (5 times the bet)
cherries  cherries  [not cherries] pays 15 × bet
cherries  cherries  cherries pays 30 × bet
BAR  BAR  BAR pays 50 × bet
7  7  7 pays 100 × bet

After the pull, display the three strings regardless of the outcome.
If the user did not win, tell him/her "Sorry, you lose.".
If he won, pay him by displaying his winnings (his original bet times the winning factor from the above table).
Then, repeat the whole process by requesting another bet amount.
----------------------------------------------------------------------------------------------------------------- */
import java.util.*;
import java.lang.Math;

public class Assig2
{  
   ///////////David////////////
   public static Scanner keyboard = new Scanner(System.in);

   public static void main(String[] args)
   {
      boolean maxChecker = true;
      //Initiates bet and calls getBet to get the users first bet.
      int bet = getBet();
      while(bet != 0)
      {
         TripleString thePull = pull();
         int multiplier = getPayMultiplier(thePull);
         int winnings = bet * multiplier;
         //Calls saveWinnings in order to store your winnings in the pullWinnings array.
         maxChecker = thePull.saveWinnings(winnings);
         //Checks to make sure the user hasn't reached MAX_PULLS. Breaks the loop if they have.
         if(!maxChecker)
         {
            System.out.println("I'm sorry, it appears you've reached your limit on lever pulls.");
            break;
         }
         display(thePull, winnings);
         bet = getBet();
      }
      
      System.out.println("Thanks for playing at the Casino!");
      
      //Creates an accessor in order to use displayWinnings.
      TripleString accessor = new TripleString();
      //Prints the pullWinnings integer array.
      System.out.println(accessor.displayWinnings());
      keyboard.close();
   }

   public static int getBet()
   {
      int bet = 0;
      boolean legalBet = false;
      while (legalBet == false)
      {
         //Prompts the user to enter a bet and gives them the valid entries.
         System.out.print("How much would you like to bet (1 - 100) or 0 to quit? ");
         bet = keyboard.nextInt();
         //Verifies that the bet entered was valid. If it wasn't the loop will repeat.
         if (0 <= bet && bet <= 100)
         {
            legalBet = true;        
         }
         else
         {
            System.out.println("Please enter a valid bet.");
         }
      }
      return bet;
   }
   ///////end David/////////
   ///////Dustin////////////
   
   //Displays results of the pull and whether or not user won
   public static void display(TripleString thePull, int winnings) 
   {    
      System.out.println("whirrrrrr .... and your pull is ... ");

      //Print the three strings from pull
      System.out.println(thePull);

      //If user wins, display win message and total winnings
      if (winnings > 0)
      {
         System.out.println("Congrats, you won $" + winnings + "\n");
      }
      //If user lost, display lose message
      else
      {
         System.out.println("Sorry - you lost.\n");
      }
   }
   
   public static int getPayMultiplier(TripleString thePull) 
   {
      //Store the three strings from the TripleString object
      String str1 = thePull.getString1();
      String str2 = thePull.getString2();
      String str3 = thePull.getString3();

      //Check if first string is "cherries"
      if (str1.equals("cherries")) 
      {
         //first string equals "cherries" and second string does not equal "cherries"
         if (!str2.equals("cherries")) 
         {
            return 5;
         }
         else 
         {
            /*first and second string equal "cherries" but third string does 
            not equal "cherries"*/
            if (!str3.equals("cherries")) 
            {
               return 15;
            }
            //All three strings equal "cherries"
            else 
            {
               return 30;
            }
         }
      }
      //All three strings equal "BAR"
      else if (str1.equals("BAR") && str2.equals("BAR") && str3.equals("BAR")) 
      {
         return 50;
      }
      //All three strings equal "7"
      else if (str1.equals("7") && str2.equals("7") && str3.equals("7")) 
      {
         return 100;
      }
      //Lose condition
      else 
      {
         return 0;
      }
   }
   //////////End Dustin//////
   
   /////////Caleb///////////
   public static TripleString pull()
   {
      /* Creates a slot machine pull result.
       * 
       * From the spec: "So this method, pull() will call the next 
       * method randString() three times to get the three strings 
       * that will be stored into the TripleString object.  Once that's 
       * done, pull() just returns the TripleString object to the client 
       * and its job is done."
       */

      //Create the resulting TripleString
      TripleString pullResult = new TripleString();

      //Add a random result for each member string.
      pullResult.setString1(randString());
      pullResult.setString2(randString());
      pullResult.setString3(randString());

      return pullResult;
   }

   private static String randString()
   {
      /*
       * This returns strings based on their individual 
       * probability.
       * 
       * From the spec:
       *   BAR       1/2  (50%)
       *   cherries  1/4  (25%)
       *   space     1/8  (12.5%)
       *   7         1/8  (12.5%)
       * 
       * I model this distribution with an array of Strings
       * such that their indices correspond to a random number.
       */

      //Get a random number between 0 and 7 (inclusive).
      //I found this method in "Absolute Java" by Walter Savitch, pg. 161.
      int randomIndex = (int) (Math.random()*8);

      //Frequency Table for desired results.
      String[] resultsTable = {
            "BAR",      //0
            "BAR",      //1
            "BAR",      //2
            "BAR",      //3 (4 of 8 possibilities, or 1/2)
            "cherries", //4
            "cherries", //5 (2 of 8 possibilities, or 1/4)
            "space",    //6 (1 of 8 possibilities, or 1/8)
            "7",        //7 (1 of 8 possibilities, or 1/8)
      };

      return resultsTable[randomIndex];
   }
   //////////End Caleb//////

}

class TripleString 
{
   /*
    * This class is a representation of a slot machine.
    * 
    * It has the slot results (string1,string2,string3),
    * and a running tally of winnings.
    */

   ////////Daisy/////////

   //Max length of a private member string.
   public static final int MAX_LEN = 20;

   //Max number of times a user can pull the slot machine's lever.
   public static final int MAX_PULLS = 40;

   //Array of the winnings per lever pull.
   private static int pullWinnings[] = new int[MAX_PULLS];

   //Slot machine results.
   private String string1, string2, string3;

   /////////End Daisy//////
   /////////Dustin/////////

   //Tracker for the number if times the slot machine lever is pulled.
   private static int numPulls  = 0;

   /////////End Dustin/////
   ////////Michael/////////

   //default constructor
   public TripleString()
   {
      string1 = "";
      string2 = "";
      string3 = ""; 
   }

   //accessor for string1
   public String getString1()
   {
      return string1;
   }

   //accessor for string2
   public String getString2()
   {
      return string2;
   }

   //accessor for string3
   public String getString3()
   {
      return string3;
   }

   //mutator for string1
   public boolean setString1( String string1 )
   {
      if (validString(string1))
      {
         this.string1 = string1;
         return true;
      }
      else
      {
         return false;
      }
   }

   //mutator for string2
   public boolean setString2( String string2 )
   {
      if(validString(string2))
      {
         this.string2 = string2;
         return true;
      }
      else
      {
         return false;
      }
   }

   //mutator for string3
   public boolean setString3( String string3 )
   {
      if (validString(string3))
      {
         this.string3 = string3;
         return true;
      }
      else
      {
         return false;
      }
   }

   //method to return all strings as one string
   public String toString()
   {
      return (" " + string1 + " " + string2 + " " + string3 + " ");
   }

   ////////////End Michael//////
   ////////////Daisy////////////
   
   // a helper function to test if a string passed to it is valid
   private boolean validString( String str )
   {
      if(str == null || str.length() > MAX_LEN)
      {
         return false;
      }
         
      return true;
   }
   // takes integer of winnings and stores it into pullWinnings
   public boolean saveWinnings(int winnings)
   {
      if (numPulls < MAX_PULLS)
      {
         pullWinnings[numPulls] = winnings;
         numPulls++;
         
         return true;
      }
      return false;
   }

   
   ////////////End Daisy////////
   
   //////////Caleb//////////////

   String displayWinnings()
   {
      /*
       * From spec: 'and the other will use a loop to get the values 
       * out of the array and return a string'
       * 
       * The format was derived from this sample output:
       * "Your individual winnings were:
       * 25 0 0 25 25 150 0 25 0 0 0 0 0 75 0"
       * Your total winnings were: $325"
       */

      //String to collect winnings.
      String output = "Your individual winnings were:\n";

      //Integer to collect total winnings
      int totalWinnings = 0;

      //Only return winnings for the number times the lever was pulled.
      for (int i = 0; i < numPulls; i++)
      {
         //Add winnings to output string, using format from example output.
         output += pullWinnings[i];

         totalWinnings += pullWinnings[i];

         if (i+1 >= numPulls)
         {
            output += "\n";
         }
         else
         {
            output += " ";
         }
      }

      return output + "Your total winnings were: $" + totalWinnings;
   }
   
   //////////End Caleb//////////
}
